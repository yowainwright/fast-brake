## Quick Start

Get up and running with fast-brake in under 2 minutes:

### 1. Installation

```bash
npm install fast-brake
# or
yarn add fast-brake
# or
bun add fast-brake
```

### 2. Basic Usage

```javascript
import { fastBrake, detect, check, getMinimumESVersion } from 'fast-brake';

// Check if code is compatible with ES5
const isES5Compatible = check('const x = () => {}', { target: 'es5' });
console.log(isES5Compatible); // false

// Get minimum ES version required
const minVersion = getMinimumESVersion('const x = () => {}');
console.log(minVersion); // 'es2015'

// Detect all ES features
const features = detect('const x = () => {}');
console.log(features);
// [{ name: 'arrow_functions', version: 'es2015', line: 1, column: 11 }]
```

### 3. That's it!

fast-brake works out of the box with zero configuration. No setup files, no complex configuration - just install and use.

## Integration Examples

### Build Tools

#### Webpack Integration

```javascript
// webpack.config.js
const { check } = require('fast-brake');

module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        use: [
          {
            loader: 'babel-loader',
            options: {
              presets: [
                ['@babel/preset-env', {
                  targets: 'es5'
                }]
              ]
            }
          },
          {
            loader: 'fast-brake-loader',
            options: {
              target: 'es5',
              mode: 'full'
            }
          }
        ]
      }
    ]
  }
};

// Custom loader (webpack-loader/fast-brake-loader.js)
const { check, detect } = require('fast-brake');

module.exports = function(source) {
  const options = this.getOptions();
  const target = options.target || 'es5';
  const mode = options.mode || 'full';
  
  const isCompatible = check(source, { 
    target, 
    quick: mode === 'quick' 
  });
  
  if (!isCompatible) {
    const features = detect(source, { target, quick: mode === 'quick' });
    const incompatible = features.filter(f => 
      getVersionIndex(f.version) > getVersionIndex(target)
    );
    
    this.emitWarning(new Error(
      `File contains incompatible ES features: ${
        incompatible.map(f => f.name).join(', ')
      }`
    ));
  }
  
  return source;
};
```

          const incompatible = features.filter(f =>

            console.warn(`${id} contains modern ES features:`,

        return null; // Don't transform, just analyze
      }
    }
  ]

### Linting Integration

#### ESLint Plugin

```javascript
// .eslintrc.js
module.exports = {
  plugins: ['fast-brake'],
  rules: {
    'fast-brake/es-version': ['error', { target: 'es2018' }]
  }
};

// eslint-plugin-fast-brake/index.js
const { detect } = require('fast-brake');

module.exports = {
  rules: {
    'es-version': {
      create(context) {
        const target = context.options[0]?.target || 'es5';
        
        return {
          Program(node) {
            const sourceCode = context.getSourceCode();
            const text = sourceCode.getText();
            
            const features = detect(text, { target });
            const incompatible = features.filter(f => 
              getVersionIndex(f.version) > getVersionIndex(target)
            );
            
            incompatible.forEach(feature => {
              context.report({
                node,
                message: `ES feature "${feature.name}" requires ${feature.version} but target is ${target}`,
                loc: {
                  line: feature.line,
                  column: feature.column
                }
              });
            });
          }
        };
      }
    }
  }
};
```

                const filePath = path.join(dir, file.name);
                
                if (file.isDirectory()) {
                  checkDirectory(filePath);
                } else if (file.name.endsWith('.js') || file.name.endsWith('.ts')) {

                    const code = fs.readFileSync(filePath, 'utf-8');

              console.error('\nâŒ ES compatibility check failed');

declare module 'fast-brake' {
  export interface DetectedFeature {
    name: string;
    version: string;
    line?: number;
    column?: number;
    snippet?: string;
  }
  
  export interface DetectionOptions {
    target: string;
    quick?: boolean;
    throwOnFirst?: boolean;
  }
  
  export function fastBrake(code: string, options: DetectionOptions): void;
  export function detect(code: string, options?: Partial<DetectionOptions>): DetectedFeature[];
  export function check(code: string, options: DetectionOptions): boolean;
  export function getMinimumESVersion(code: string, options?: { quick?: boolean }): string;
}
